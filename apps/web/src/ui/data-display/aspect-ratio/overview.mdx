import { Canvas, Controls, Meta, Subtitle, Title } from '@storybook/blocks';

import * as AspectRatioStories from './index.stories';

<Meta of={AspectRatioStories} name='Overview' />
<Title>Aspect Ratio</Title>
<Subtitle>Displays content within a desired ratio.</Subtitle>

## Props

<Canvas of={AspectRatioStories.Default} />
<Controls of={AspectRatioStories.Default} />

```jsx
import Image from 'next/image';

import { AspectRatio } from '#/ui/data-display';

export const AspectRatioDemo = () => (
  <AspectRatio ratio={16 / 9} className='bg-muted'>
    <Image
      src='https://images.unsplash.com/photo-1588345921523-c2dcdb7f1dcd?w=800&dpr=2&q=80'
      alt='Photo by Drew Beamer'
      fill
      className='rounded-md object-cover'
    />
  </AspectRatio>
);
```

## Usage

```jsx
import Image from 'next/image';

import { AspectRatio } from '#/ui/data-display';

<div className='w-[450px]'>
  <AspectRatio ratio={16 / 9}>
    <Image src='...' alt='Image' className='rounded-md object-cover' />
  </AspectRatio>
</div>;
```
