import {Meta, Typeset} from "@storybook/blocks";
import {notoSansJP, inter, newsreader, jetBrainsMono} from "#/lib/font";
import defaultTheme from "tailwindcss/defaultTheme";

export const remToPx = (rem, baseFontSize = 16) => {
    if (typeof rem === 'string') {
        rem = parseFloat(rem.replace('rem', ''));
    }
    return rem * baseFontSize;
};

export const typography = {
    type: {
        notoSansJP: notoSansJP.style.fontFamily,
        inter: inter.style.fontFamily,
        newsreader: newsreader.style.fontFamily,
        jetBrainsMono: jetBrainsMono.style.fontFamily,
    },
    weight: {
        regular: defaultTheme.fontWeight.normal,
        medium: defaultTheme.fontWeight.medium,
    },
    size: {
        xs: remToPx(defaultTheme.fontSize.xs[0]),
        sm: remToPx(defaultTheme.fontSize.sm[0]),
        base: remToPx(defaultTheme.fontSize.base[0]),
        lg: remToPx(defaultTheme.fontSize.lg[0]),
        xl: remToPx(defaultTheme.fontSize.xl[0]),
        '2xl': remToPx(defaultTheme.fontSize['2xl'][0]),
    },
};

export const SampleTextEn = 'Recognizing the inherent dignity and the equal and inalienable rights of all members of the human family';
